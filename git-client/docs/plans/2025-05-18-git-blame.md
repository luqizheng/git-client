# Git Blame 功能实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现 Git Blame 功能，支持在代码编辑器中显示每行代码的最后修改者信息（作者、提交哈希、时间）。

**Architecture:** 后端使用 git2 库提供 blame 数据查询接口，前端在 Monaco Editor 的左侧边栏（glyph margin）渲染 blame 信息，支持点击跳转提交详情。

**Tech Stack:** Rust (git2) + Vue 3 + TypeScript + Monaco Editor + Tauri IPC

```rust
// src-tauri/src/services/blame_service.rs
use crate::models::blame::{BlameLine, BlameResult};
use crate::utils::error::AppError;
use git2::{BlameOptions, Repository};

pub fn blame_file(
    repo: &Repository,
    file_path: &str,
    commit_id: Option<&str>,
) -> Result<BlameResult, AppError> {
    let mut opts = BlameOptions::new();
    
    // 如果指定了 commit，从该 commit 开始 blame
    if let Some(cid) = commit_id {
        let oid = git2::Oid::from_str(cid)?;
        opts.newest_commit(oid);
    }
    
    // 检查文件是否存在
    let full_path = repo.workdir()
        .map(|w| w.join(file_path))
        .filter(|p| p.exists());
    
    if full_path.is_none() && commit_id.is_none() {
        return Err(AppError::Git(format!("File not found: {}", file_path)));
    }
    
    let blame = repo.blame_file(std::path::Path::new(file_path), Some(&mut opts))?;
    let mut lines = Vec::new();
    
    // 遍历每个 hunk，每个 hunk 包含多行
    for hunk in blame.iter() {
        let commit = repo.find_commit(hunk.final_commit_id())?;
        let author = commit.author();
        
        // 展开 hunk 中的每一行
        for line_offset in 0..hunk.lines_in_hunk() {
            lines.push(BlameLine {
                line_number: hunk.final_start_line() + line_offset,
                commit_id: hunk.final_commit_id().to_string(),
                author: author.name().unwrap_or("Unknown").to_string(),
                author_email: author.email().unwrap_or("").to_string(),
                timestamp: author.when().seconds(),
                summary: commit.summary().unwrap_or("").to_string(),
                is_boundary: hunk.is_boundary(),
            });
        }
    }
    
    // 按行号排序确保顺序正确
    lines.sort_by_key(|l| l.line_number);
    
    Ok(BlameResult {
        file_path: file_path.to_string(),
        lines,
    })
}
```

```rust
use crate::models::blame::BlameResult;
use crate::services::blame_service;
use crate::utils::error::AppError;
use crate::AppState;
use tauri::State;

#[tauri::command]
pub async fn blame_file(
    state: State<'_, AppState>,
    repo_path: String,
    file_path: String,
    commit_id: Option<String>,
) -> Result<BlameResult, AppError> {
    let repos = state.repos.clone();
    tokio::task::spawn_blocking(move || {
        let manager = repos.lock().map_err(|e| AppError::Credential(e.to_string()))?;
        let repo = manager.get_repo(&repo_path)?;
        blame_service::blame_file(&repo, &file_path, commit_id.as_deref())
    })
    .await?
}
```

### Step 1: 创建 store

```typescript
// src/stores/blame.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { BlameResult, BlameLine } from '../types/git'
import { blameFile } from '../utils/ipc'

export const useBlameStore = defineStore('blame', () => {
  // State
  const blameData = ref<Map<string, BlameResult>>(new Map())
  const loading = ref(false)
  const error = ref<string | null>(null)
  const visible = ref(false)
  
  // Getters
  const getBlameForFile = computed(() => (repoPath: string, filePath: string) => {
    const key = `${repoPath}:${filePath}`
    return blameData.value.get(key) || null
  })
  
  const isLoading = computed(() => loading.value)
  const isVisible = computed(() => visible.value)
  
  // Actions
  async function fetchBlame(repoPath: string, filePath: string, commitId?: string) {
    const key = `${repoPath}:${filePath}`
    loading.value = true
    error.value = null
    
    try {
      const result = await blameFile(repoPath, filePath, commitId)
      blameData.value.set(key, result)
      return result
    } catch (e) {
      error.value = String(e)
      throw e
    } finally {
      loading.value = false
    }
  }
  
  function clearBlame(repoPath: string, filePath: string) {
    const key = `${repoPath}:${filePath}`
    blameData.value.delete(key)
  }
  
  function toggleVisibility() {
    visible.value = !visible.value
  }
  
  function show() {
    visible.value = true
  }
  
  function hide() {
    visible.value = false
  }
  
  return {
    blameData,
    loading,
    error,
    visible,
    getBlameForFile,
    isLoading,
    isVisible,
    fetchBlame,
    clearBlame,
    toggleVisibility,
    show,
    hide,
  }
})
```

```rust
// 在 blame_file 函数开头添加
fn is_binary_file(repo: &Repository, file_path: &str, commit_id: Option<&str>) -> Result<bool, AppError> {
    // 尝试读取文件内容检测是否为二进制
    let content = if let Some(cid) = commit_id {
        let oid = git2::Oid::from_str(cid)?;
        let commit = repo.find_commit(oid)?;
        let tree = commit.tree()?;
        match tree.get_path(std::path::Path::new(file_path)) {
            Ok(entry) => {
                let blob = repo.find_blob(entry.id())?;
                blob.content().to_vec()
            }
            Err(_) => return Ok(false),
        }
    } else {
        match repo.workdir() {
            Some(w) => std::fs::read(w.join(file_path)).unwrap_or_default(),
            None => return Ok(false),
        }
    };
    
    // 检测是否包含 null 字节
    Ok(content.contains(&0))
}
```
